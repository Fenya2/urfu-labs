def _split(a, b, h):
    """
    :param a:  
    :param b:  
    :param h: 
    :return:    [a, b] c  h
    """
    n = int((b - a) / h)
    return [a + h * i for i in range(n + 1)]


def avg_rectangles(f, a, b, h):
    """
    :param f:  
    :param a:  
    :param b:  
    :param h:   
    :return:       [a, b]      h,
         
    """
    return h * sum(f(x) for x in _split(a, b, h / 2)[1::2])


def trapeze(f, a, b, h):
    """
    :param f:  
    :param a:  
    :param b:  
    :param h:   
    :return:       [a, b]      h,
        
    """
    Y = [f(x) for x in _split(a, b, h)]
    sum = 0
    for i in range(1, len(Y)):
        sum += Y[i - 1] * Y[i]
    return h / 2 * sum


def trapeze_opt(f, a, b, h):
    """
    :param f:  
    :param a:  
    :param b:  
    :param h:   
    :return:       [a, b]      h,
         (, ,   )
    """
    Y = [f(x) for x in _split(a, b, h)]
    return h / 2 * (Y[0] + Y[-1] + 2 * sum(Y[1:-1]))


def simpson(f, a, b, h):
    """
    :param f:  
    :param a:  
    :param b:  
    :param h:   
    :return:       [a, b]      h,
        
    """
    Y = [f(x) for x in _split(a, b, h / 2)]
    sum = 0
    for i in range(2, len(Y), 2):
        sum += Y[i - 2] + 4 * Y[i - 1] + Y[i]
    return h / 6 * sum
